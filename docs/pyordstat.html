10class FiniteOrderStatistics(BaseOrderStatistics):
11    """Order statistics distribution for discrete distributions with finite support."""
12
13    _x: NDArray[np.number]
14    _pdf: NDArray[np.number]
15    _cdf: NDArray[np.number]
16
17    def __init__(self, x: NDArray[np.number], pmf: NDArray[np.number]) -> None:
18        """Create a discrete order statistics distribution.
19
20        Requires the values of the random variable and the probability mass function.
21        The x values can be in any order, but the pdf values must be in the same order
22        as the x values, and they will be sorted in ascending order.
23
24        Args:
25            x (NDArray[np.number]): Support of the distribution.
26            pdf (NDArray[np.number]): Probability mass function of the distribution.
27        """
28        isort = np.argsort(x)
29        self._x = np.asarray(x)[isort]
30        pmf = np.asarray(pmf)[isort]
31        cdf = np.cumsum(pmf)
32
33        # Normalize
34        pmf = pmf / cdf[-1]
35        cdf = cdf / cdf[-1]
36
37        super().__init__(pmf, cdf)
38
39    @property
40    def x(self) -> NDArray[np.number]:
41        """Support of the distribution."""
42        return self._x
43
44    @property
45    def pdf(self) -> NDArray[np.number]:
46        """Probability density function of order statistics."""
47        raise NotImplementedError("PDF not implemented for discrete order statistics.")
48
49    @property
50    def pmf(self) -> NDArray[np.number]:
51        """Probability mass function of order statistics."""
52        return self._pdf
53
54    @property
55    def cdf(self) -> NDArray[np.number]:
56        """Cumulative distribution function."""
57        return self._cdf
58
59    def order_statistic_pmf(self, n: int, k: int) -> NDArray[np.number]:
60        """Order statistic probability mass function.
61
62        Args:
63            n (int): Sample size.
64            k (int): Order statistic to calculate.
65
66        Returns
67        -------
68            NDArray[np.number]: Probability mass function of the k-th order statistic.
69        """
70        cdf = self.order_statistic_cdf(n, k)
71        return np.diff(cdf, prepend=0)
72
73    def order_statistic_cdf(self, n: int, k: int) -> NDArray[np.number]:

78            k (int): Order statistic to calculate.
79
80        Returns
81        -------
82            NDArray[np.number]: Cumulative distribution function of the k-th order statistic.
83        """
84        return ordstat_cdf(self._cdf, n, k)

51    def order_statistic_pmf(self, x: NDArray[np.number], n: int, k: int) -> NDArray[np.number]:
52        """Calculate the k-th order statistic PMF of a sample of size n.
53
54        Return the k-th order statistic probability mass function of a sample of size n
55        from a discrete distribution, given the values in x.
56
57        Args:
58            x (NDArray[np.number]): Values to evaluate the PMF at.
59            n (int): Sample size.
60            k (int): Order statistic.
61
62        Returns
63        -------
64            NDArray[np.number]: PMF values.
65        """
66        cdf_1 = self.order_statistic_cdf(x, n, k)
67        cdf_0 = self.order_statistic_cdf(x - 1, n, k)

69        return cdf_1 - cdf_0
